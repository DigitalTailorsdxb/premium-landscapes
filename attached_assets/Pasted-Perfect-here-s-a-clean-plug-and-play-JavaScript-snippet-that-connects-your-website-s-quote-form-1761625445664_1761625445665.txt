üíª Frontend JavaScript (for your Replit site)
<script>
document.addEventListener("DOMContentLoaded", () => {
  const form = document.getElementById("quoteForm"); // your form's ID
  const messageBox = document.getElementById("formMessage"); // optional message div

  form.addEventListener("submit", async (e) => {
    e.preventDefault();

    // Build JSON payload from form fields
    const formData = new FormData(form);
    const data = Object.fromEntries(formData.entries());

    // Show loading message
    if (messageBox) {
      messageBox.textContent = "Generating your quote...";
      messageBox.style.color = "#555";
    }

    try {
      // Send data to n8n webhook
      const response = await fetch("YOUR_N8N_WEBHOOK_URL", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(data),
      });

      const result = await response.json();

      // Handle success / error response from n8n
      if (result.success) {
        messageBox.textContent = "‚úÖ " + result.message;
        messageBox.style.color = "green";
        form.reset();
      } else {
        messageBox.textContent = "‚ö†Ô∏è Something went wrong ‚Äî please try again.";
        messageBox.style.color = "red";
      }
    } catch (err) {
      console.error("Error:", err);
      if (messageBox) {
        messageBox.textContent = "‚ùå Network error ‚Äî please try again.";
        messageBox.style.color = "red";
      }
    }
  });
});
</script>
